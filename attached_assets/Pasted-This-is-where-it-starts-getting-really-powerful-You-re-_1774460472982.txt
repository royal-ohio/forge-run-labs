## 📁 `server/utils.py` (NEW)

```python
import re
import datetime

BASE_DOMAIN = "forgerunlabs.com"

def generate_slug(name: str):
    slug = name.lower()
    slug = re.sub(r'[^a-z0-9]+', '-', slug)
    return slug.strip('-')

def build_subdomain(slug: str):
    return f"{slug}.{BASE_DOMAIN}"

def timestamp():
    return datetime.datetime.utcnow().isoformat()
```

## 📁 `server/main.py` (UPDATED CORE LOGIC)

Replace your create endpoint with this:

```python
from utils import generate_slug, build_subdomain, timestamp
import requests

@app.post("/api/admin/projects")
def create_project(project: dict, db: Session = Depends(get_db), token: str = Depends(verify_admin)):
    
    slug = generate_slug(project["name"])
    subdomain = build_subdomain(slug)

    url = f"https://{subdomain}"

    new_project = Project(
        name=project["name"],
        description=project.get("description"),
        slug=slug,
        subdomain=subdomain,
        url=url,
        deploy_url=project.get("deploy_url"),
        status="BUILDING"
    )

    db.add(new_project)
    db.commit()
    db.refresh(new_project)

    return new_project
```

# 🚀 DEPLOYMENT HOOK ENDPOINT

## 📁 Add this to `server/main.py`

```python
@app.post("/api/admin/deploy/{project_id}")
def trigger_deploy(project_id: int, db: Session = Depends(get_db), token: str = Depends(verify_admin)):
    
    project = db.query(Project).get(project_id)

    if not project or not project.deploy_url:
        raise HTTPException(400, "Deploy URL not configured")

    try:
        res = requests.post(project.deploy_url, timeout=10)

        project.status = "LIVE" if res.status_code == 200 else "FAILED"
        project.last_deployed_at = timestamp()

        db.commit()

        return {
            "status": project.status,
            "response_code": res.status_code
        }

    except Exception as e:
        project.status = "FAILED"
        db.commit()

        return {"error": str(e)}
```

# 💓 HEALTH CHECK SYSTEM

## 📁 Add this endpoint

```python
@app.get("/api/health/{project_id}")
def check_health(project_id: int, db: Session = Depends(get_db)):
    
    project = db.query(Project).get(project_id)

    try:
        res = requests.get(project.url, timeout=5)

        project.health = "HEALTHY" if res.status_code == 200 else "DEGRADED"
        db.commit()

        return {"health": project.health}

    except:
        project.health = "OFFLINE"
        db.commit()
        return {"health": "OFFLINE"}
```
