## 📁 `server/models.py`

```python
from sqlalchemy import Column, Integer, String, Boolean
from db import Base

class Project(Base):
    __tablename__ = "projects"

    id = Column(Integer, primary_key=True, index=True)
    name = Column(String)
    description = Column(String)
    subdomain = Column(String)
    url = Column(String)
    status = Column(String, default="LIVE")  # LIVE / BUILDING / OFFLINE
    is_public = Column(Boolean, default=True)
```

## 📁 `server/main.py`

```python
from fastapi import FastAPI, Depends, HTTPException, Header
from sqlalchemy.orm import Session
from db import SessionLocal, engine, Base
from models import Project

import json

app = FastAPI()

Base.metadata.create_all(bind=engine)

def get_db():
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()

# --- Load Admin Token ---
with open("config/settings.json") as f:
    SETTINGS = json.load(f)

ADMIN_TOKEN = SETTINGS["admin_token"]

def verify_admin(token: str = Header(None)):
    if token != ADMIN_TOKEN:
        raise HTTPException(status_code=403, detail="Unauthorized")

# --- Public API ---
@app.get("/api/projects")
def get_projects(db: Session = Depends(get_db)):
    return db.query(Project).filter(Project.is_public == True).all()

# --- Admin APIs ---
@app.post("/api/admin/projects")
def create_project(project: dict, db: Session = Depends(get_db), token: str = Depends(verify_admin)):
    new_project = Project(**project)
    db.add(new_project)
    db.commit()
    db.refresh(new_project)
    return new_project

@app.get("/api/admin/projects")
def admin_list(db: Session = Depends(get_db), token: str = Depends(verify_admin)):
    return db.query(Project).all()

@app.delete("/api/admin/projects/{project_id}")
def delete_project(project_id: int, db: Session = Depends(get_db), token: str = Depends(verify_admin)):
    proj = db.query(Project).get(project_id)
    if not proj:
        raise HTTPException(404, "Not found")

    db.delete(proj)
    db.commit()
    return {"status": "deleted"}
```

# 🛠️ ADMIN PANEL

## 📁 `client/admin.html`

```html
<!DOCTYPE html>
<html>
<head>
  <title>Admin - ForgeRun Labs</title>
</head>
<body style="background:#111;color:white;font-family:sans-serif;padding:40px;">

<h1>Admin Panel</h1>

<input id="token" placeholder="Admin Token" style="width:300px;"><br><br>

<h2>Add Project</h2>

<input id="name" placeholder="Name"><br>
<input id="desc" placeholder="Description"><br>
<input id="url" placeholder="URL"><br>
<input id="status" placeholder="Status (LIVE)"><br><br>

<button onclick="addProject()">Create</button>

<h2>Projects</h2>
<div id="list"></div>

<script>
async function addProject() {
  const token = document.getElementById("token").value;

  const body = {
    name: document.getElementById("name").value,
    description: document.getElementById("desc").value,
    url: document.getElementById("url").value,
    status: document.getElementById("status").value || "LIVE"
  };

  await fetch("/api/admin/projects", {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      "token": token
    },
    body: JSON.stringify(body)
  });

  loadProjects();
}

async function loadProjects() {
  const token = document.getElementById("token").value;

  const res = await fetch("/api/admin/projects", {
    headers: { token }
  });

  const data = await res.json();

  const list = document.getElementById("list");
  list.innerHTML = "";

  data.forEach(p => {
    const el = document.createElement("div");
    el.innerHTML = `
      ${p.name} - ${p.status}
      <button onclick="deleteProject(${p.id})">Delete</button>
    `;
    list.appendChild(el);
  });
}

async function deleteProject(id) {
  const token = document.getElementById("token").value;

  await fetch(`/api/admin/projects/${id}`, {
    method: "DELETE",
    headers: { token }
  });

  loadProjects();
}
</script>

</body>
</html>
```
